44  public static float Generate(float x)
45  {
46  int i0 = FastFloor(x);
47  int i1 = i0 + 1;
48  float x0 = x - i0;
49  float x1 = x0 - 1.0f;
50 
51  float n0, n1;

53  float t0 = 1.0f - x0 * x0;

71  public static float Generate(float x, float y)
72  {
73  const float F2 = 0.366025403f; //* F2 = 0.5*(sqrt(3.0)-1.0)
74  const float G2 = 0.211324865f; //* G2 = (3.0-Math.sqrt(3.0))/6.0
75 
76  float n0, n1, n2; //* Noise contributions from the three corners
77 
78  //* Skew the input space to determine which simplex cell we're in
79  float s = (x + y) * F2; //* Hairy factor for 2D
80  float xs = x + s;
81  float ys = y + s;
82  int i = FastFloor(xs);
83  int j = FastFloor(ys);
84 
85  float t = (float)(i + j) * G2;
86  float X0 = i - t; //* Unskew the cell origin back to (x,y) space
87  float Y0 = j - t;
88  float x0 = x - X0; //* The x,y distances from the cell origin
89  float y0 = y - Y0;
90 
91  //* For the 2D case, the simplex shape is an equilateral triangle.
92  //* Determine which simplex we are in.
93  int i1, j1; //* Offsets for second (middle) corner of simplex in (i,j) coords
94  if (x0 > y0) { i1 = 1; j1 = 0; } //* lower triangle, XY order: (0,0)->(1,0)->(1,1)
95  else { i1 = 0; j1 = 1; } //* upper triangle, YX order: (0,0)->(0,1)->(1,1)
96 
97  //* A step of (1,0) in (i,j) means a step of (1-c,-c) in (x,y), and
98  //* a step of (0,1) in (i,j) means a step of (-c,1-c) in (x,y), where
99  //* c = (3-sqrt(3))/6
100 
101  float x1 = x0 - i1 + G2; //* Offsets for middle corner in (x,y) unskewed coords
102  float y1 = y0 - j1 + G2;
103  float x2 = x0 - 1.0f + 2.0f * G2; //* Offsets for last corner in (x,y) unskewed coords
104  float y2 = y0 - 1.0f + 2.0f * G2;
105 
106  //* Wrap the integer indices at 256, to avoid indexing perm[] out of bounds
107  int ii = i % 256;
108  int jj = j % 256;
109 
110  //* Calculate the contribution from the three corners
111  float t0 = 0.5f - x0 * x0 - y0 * y0;
112  if (t0 < 0.0f) n0 = 0.0f;
113  else
114  {
115  t0 *= t0;
116  n0 = t0 * t0 * grad(perm[ii + perm[jj]], x0, y0);
117  }
118 
119  float t1 = 0.5f - x1 * x1 - y1 * y1;
120  if (t1 < 0.0f) n1 = 0.0f;
121  else
122  {
123  t1 *= t1;
124  n1 = t1 * t1 * grad(perm[ii + i1 + perm[jj + j1]], x1, y1);
125  }
126 
127  float t2 = 0.5f - x2 * x2 - y2 * y2;
128  if (t2 < 0.0f) n2 = 0.0f;
129  else
130  {
131  t2 *= t2;
132  n2 = t2 * t2 * grad(perm[ii + 1 + perm[jj + 1]], x2, y2);
133  }
134 
135  //* Add contributions from each corner to get the final noise value.
136  //* The result is scaled to return values in the interval [-1,1].
137  return 40.0f * (n0 + n1 + n2); //* TODO: The scale factor is preliminary!
138  }

141  public static float Generate(float x, float y, float z)
142  {
143  //* Simple skewing factors for the 3D case
144  const float F3 = 0.333333333f;
145  const float G3 = 0.166666667f;
146 
147  float n0, n1, n2, n3; //* Noise contributions from the four corners
148 
149  //* Skew the input space to determine which simplex cell we're in
150  float s = (x + y + z) * F3; //* Very nice and simple skew factor for 3D
151  float xs = x + s;
152  float ys = y + s;
153  float zs = z + s;
154  int i = FastFloor(xs);
155  int j = FastFloor(ys);
156  int k = FastFloor(zs);
157 
158  float t = (float)(i + j + k) * G3;
159  float X0 = i - t; //* Unskew the cell origin back to (x,y,z) space
160  float Y0 = j - t;
161  float Z0 = k - t;
162  float x0 = x - X0; //* The x,y,z distances from the cell origin
163  float y0 = y - Y0;
164  float z0 = z - Z0;
165 
166  //* For the 3D case, the simplex shape is a slightly irregular tetrahedron.
167  //* Determine which simplex we are in.
168  int i1, j1, k1; //* Offsets for second corner of simplex in (i,j,k) coords
169  int i2, j2, k2; //* Offsets for third corner of simplex in (i,j,k) coords
170 
171  /* This code would benefit from a backport from the GLSL version! */
172  if (x0 >= y0)
173  {
174  if (y0 >= z0)
175  { i1 = 1; j1 = 0; k1 = 0; i2 = 1; j2 = 1; k2 = 0; } //* X Y Z order
176  else if (x0 >= z0) { i1 = 1; j1 = 0; k1 = 0; i2 = 1; j2 = 0; k2 = 1; } //* X Z Y order
177  else { i1 = 0; j1 = 0; k1 = 1; i2 = 1; j2 = 0; k2 = 1; } //* Z X Y order
178  }
179  else
180  { //* x0<y0
181  if (y0 < z0) { i1 = 0; j1 = 0; k1 = 1; i2 = 0; j2 = 1; k2 = 1; } //* Z Y X order
182  else if (x0 < z0) { i1 = 0; j1 = 1; k1 = 0; i2 = 0; j2 = 1; k2 = 1; } //* Y Z X order
183  else { i1 = 0; j1 = 1; k1 = 0; i2 = 1; j2 = 1; k2 = 0; } //* Y X Z order
184  }
185 
186  //* A step of (1,0,0) in (i,j,k) means a step of (1-c,-c,-c) in (x,y,z),
187  //* a step of (0,1,0) in (i,j,k) means a step of (-c,1-c,-c) in (x,y,z), and
188  //* a step of (0,0,1) in (i,j,k) means a step of (-c,-c,1-c) in (x,y,z), where
189  //* c = 1/6.
190 
191  float x1 = x0 - i1 + G3; //* Offsets for second corner in (x,y,z) coords
192  float y1 = y0 - j1 + G3;
193  float z1 = z0 - k1 + G3;
194  float x2 = x0 - i2 + 2.0f * G3; //* Offsets for third corner in (x,y,z) coords
195  float y2 = y0 - j2 + 2.0f * G3;
196  float z2 = z0 - k2 + 2.0f * G3;
197  float x3 = x0 - 1.0f + 3.0f * G3; //* Offsets for last corner in (x,y,z) coords
198  float y3 = y0 - 1.0f + 3.0f * G3;
199  float z3 = z0 - 1.0f + 3.0f * G3;
200 
201  //* Wrap the integer indices at 256, to avoid indexing perm[] out of bounds
202  int ii = Mod(i, 256);
203  int jj = Mod(j, 256);
204  int kk = Mod(k, 256);
205 
206  //* Calculate the contribution from the four corners
207  float t0 = 0.6f - x0 * x0 - y0 * y0 - z0 * z0;
208  if (t0 < 0.0f) n0 = 0.0f;
209  else
210  {
211  t0 *= t0;
212  n0 = t0 * t0 * grad(perm[ii + perm[jj + perm[kk]]], x0, y0, z0);
213  }
214 
215  float t1 = 0.6f - x1 * x1 - y1 * y1 - z1 * z1;
216  if (t1 < 0.0f) n1 = 0.0f;
217  else
218  {
219  t1 *= t1;
220  n1 = t1 * t1 * grad(perm[ii + i1 + perm[jj + j1 + perm[kk + k1]]], x1, y1, z1);
221  }
222 
223  float t2 = 0.6f - x2 * x2 - y2 * y2 - z2 * z2;
224  if (t2 < 0.0f) n2 = 0.0f;
225  else
226  {
227  t2 *= t2;
228  n2 = t2 * t2 * grad(perm[ii + i2 + perm[jj + j2 + perm[kk + k2]]], x2, y2, z2);
229  }
230 
231  float t3 = 0.6f - x3 * x3 - y3 * y3 - z3 * z3;
232  if (t3 < 0.0f) n3 = 0.0f;
233  else
234  {
235  t3 *= t3;
236  n3 = t3 * t3 * grad(perm[ii + 1 + perm[jj + 1 + perm[kk + 1]]], x3, y3, z3);
237  }
238 
239  //* Add contributions from each corner to get the final noise value.
240  //* The result is scaled to stay just inside [-1,1]
241  return 32.0f * (n0 + n1 + n2 + n3); //* TODO: The scale factor is preliminary!
242  }

283  private static float grad(int hash, float x)
284  {
285  int h = hash & 15;
286  float grad = 1.0f + (h & 7); //* Gradient value 1.0, 2.0, ..., 8.0
287  if ((h & 8) != 0) grad = -grad; //* Set a random sign for the gradient
288  return (grad * x); //* Multiply the gradient with the distance
289  }

291  private static float grad(int hash, float x, float y)
292  {
293  int h = hash & 7; //* Convert low 3 bits of hash code
294  float u = h < 4 ? x : y; //* into 8 simple gradient directions,
295  float v = h < 4 ? y : x; //* and compute the dot product with (x,y).
296  return ((h & 1) != 0 ? -u : u) + ((h & 2) != 0 ? -2.0f * v : 2.0f * v);
297  }
298 
299  private static float grad(int hash, float x, float y, float z)
300  {
301  int h = hash & 15; //* Convert low 4 bits of hash code into 12 simple
302  float u = h < 8 ? x : y; //* gradient directions, and compute dot product.
303  float v = h < 4 ? y : h == 12 || h == 14 ? x : z; //* Fix repeats at h = 12 to 15
304  return ((h & 1) != 0 ? -u : u) + ((h & 2) != 0 ? -v : v);
305  }
306 
307  private static float grad(int hash, float x, float y, float z, float t)
308  {
309  int h = hash & 31; //* Convert low 5 bits of hash code into 32 simple
310  float u = h < 24 ? x : y; //* gradient directions, and compute dot product.
311  float v = h < 16 ? y : z;
312  float w = h < 8 ? z : t;
313  return ((h & 1) != 0 ? -u : u) + ((h & 2) != 0 ? -v : v) + ((h & 4) != 0 ? -w : w);
314  }
